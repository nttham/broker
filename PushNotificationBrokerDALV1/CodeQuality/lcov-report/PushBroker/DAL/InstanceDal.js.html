var InstanceModel = require('../models/Instances.js');
var db;
var connector = require('./Connector.js');
 
//this api will insert the data to Instances collection
exports.saveData = function(obj,callback){
    db = connector.getDB();
    var Instance = db.model('Instances', InstanceModel);
    var instanceDal = new Instance(obj);
    instanceDal.save(callback);
 
};
 
//this api will fetch the info of instances from the DB
exports.getInstance = function(obj,callback){
    db = connector.getDB();
    var Instance = db.model('Instances', InstanceModel);
 
    Instance.find(obj,callback);
 
};
 
//this api will updates the Instances Collection record
exports.updateInstance = function(conditions, updateobj,callback){
    db = connector.getDB();
    var Instance = db.model('Instances', InstanceModel);
 
    Instance.update(conditions, updateobj,callback);
 
};
 
 
//this api will delete the record from instances
exports.deleteInstance = function(obj,callback){
    db = connector.getDB();
    var Instance = db.model('Instances', InstanceModel);
 
    Instance.remove(obj,callback);
 
};
